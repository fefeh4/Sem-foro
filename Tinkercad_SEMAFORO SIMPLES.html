int counter;


void setup()

{

pinMode(8, OUTPUT);
pinMode(9, OUTPUT);
pinMode(10, OUTPUT);
}


void loop()

{

for (counter = 0; counter < 10; ++counter) {
digitalWrite(8, HIGH);
digitalWrite(9, LOW);
digitalWrite(10, LOW);
delay(4000); // Wait for 4000 millisecond(s)
digitalWrite(8, LOW);
digitalWrite(9, LOW);
digitalWrite(10, HIGH);
delay(4000); // Wait for 4000 millisecond(s)
digitalWrite(8, LOW);
digitalWrite(9, HIGH);
digitalWrite(10, LOW);
delay(4000); // Wait for 4000 millisecond(s)
}
  
}
